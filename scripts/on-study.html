xw <- 
  xd %>%
  spread(key = "event", value = "date") %>%
  mutate(on.study = (`Visit 4` - as.POSIXct(Sys.Date())) > 0,
         completed = (`Visit 4` - as.POSIXct(Sys.Date())) < 0)

t.on.study <- 
  table(xw$on.study, useNA = "ifany") %>%
  data.frame

gc <-
